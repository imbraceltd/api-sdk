2. **建立 AI Agent**

   `workflow_name` 在你的組織內必須是唯一的。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const agent = await client.chatAi.createAiAgent({
         name: "Support Bot",
         workflow_name: "support_bot_v1",
         description: "處理第一線客戶支援查詢",
         instructions: "你是一個樂於助人的支援代理。請保持簡潔與友善。",
         provider_id: "system",   // 使用組織的預設 LLM 提供者
         model_id: "gpt-4o",      // 系統提供者暴露的任何模型名稱
       })

       const assistantId = agent.id  // UUID — 後續所有呼叫都使用此 ID
       console.log("AI agent created:", assistantId)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       agent = client.chat_ai.create_ai_agent({
           "name": "Support Bot",
           "workflow_name": "support_bot_v1",
           "description": "處理第一線客戶支援查詢",
           "instructions": "你是一個樂於助人的支援代理。請保持簡潔與友善。",
           "provider_id": "system",   # 使用組織的預設 LLM 提供者
           "model_id": "gpt-4o",      # 系統提供者暴露的任何模型名稱
       })

       assistant_id = agent["id"]
       print("AI agent created:", assistant_id)
       ```
     </TabItem>
   </Tabs>

   `provider_id` 與 `model_id` 為必填。傳入 `provider_id: "system"` 以委派給組織的預設 LLM 提供者，或傳入自訂提供者的 UUID。使用 `provider_id: "system"` 時，`model_id` 可接受模型名稱如 `"gpt-4o"`，或傳入字面值 `"Default"` 以使用系統預設模型。

3. **使用 AI Agent 串流聊天回應**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const response = await client.aiAgent.streamChat({
         assistant_id: assistantId,
         messages: [{ role: "user", content: "我要如何重設密碼？" }],
         // id 為工作階段 UUID — 重複使用以維持對話歷史
         // 若省略，每次呼叫會自動產生新的 UUID
       })

       const reader = response.body!.getReader()
       const decoder = new TextDecoder()

       while (true) {
         const { done, value } = await reader.read()
         if (done) break
         const text = decoder.decode(value)
         for (const line of text.split("\n")) {
           if (line.startsWith("data: ")) {
             const data = line.slice(6).trim()
             if (data && data !== "[DONE]") {
               try {
                 const chunk = JSON.parse(data)
                 process.stdout.write(chunk.delta ?? chunk.content ?? "")
               } catch {}
             }
           }
         }
       }
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       response = client.ai_agent.stream_chat({
           "assistant_id": assistant_id,
           "messages": [{"role": "user", "content": "我要如何重設密碼？"}],
           # id 為工作階段 UUID — 重複使用以維持對話歷史。
           # 若省略，每次呼叫會自動產生新的 UUID。
       })

       import json
       for line in response.iter_lines():
           if isinstance(line, bytes):
               line = line.decode()
           if not line.startswith("data: "):
               continue
           data = line[6:].strip()
           if data and data != "[DONE]":
               try:
                   chunk = json.loads(data)
                   print(chunk.get("delta") or chunk.get("content") or "", end="")
               except Exception:
                   pass
       ```
     </TabItem>
   </Tabs>

4. **維持對話歷史（工作階段 ID）**

   在多次呼叫中傳入相同的 `id`（必須是 UUID）以保持上下文：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       import { randomUUID } from "crypto"

       const sessionId = randomUUID()

       // 第一則訊息
       await client.aiAgent.streamChat({
         assistant_id: assistantId,
         id: sessionId,
         messages: [{ role: "user", content: "你們的退款政策是什麼？" }],
       })

       // 後續追問 — 相同工作階段，助理記得上下文
       await client.aiAgent.streamChat({
         assistant_id: assistantId,
         id: sessionId,
         messages: [{ role: "user", content: "需要多久時間？" }],
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       import uuid

       session_id = str(uuid.uuid4())

       # 第一則訊息
       client.ai_agent.stream_chat({
           "assistant_id": assistant_id,
           "id": session_id,
           "messages": [{"role": "user", "content": "你們的退款政策是什麼？"}],
       })

       # 後續追問 — 相同工作階段，助理記得上下文
       client.ai_agent.stream_chat({
           "assistant_id": assistant_id,
           "id": session_id,
           "messages": [{"role": "user", "content": "需要多久時間？"}],
       })
       ```
     </TabItem>
   </Tabs>

</Steps>

## 2. 建立工作流程並綁定至助理

<Steps>

1. **列出現有流程以尋找你的專案 ID**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const { data: flows } = await client.workflows.listFlows({ limit: 5 })
       const projectId = flows[0]?.projectId
       console.log("Project ID:", projectId)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       res = client.workflows.list_flows(limit=5)
       flows = res.get("data", [])
       project_id = flows[0]["projectId"] if flows else None
       print("Project ID:", project_id)
       ```
     </TabItem>
   </Tabs>

2. **建立新流程**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const flow = await client.workflows.createFlow({
         displayName: "CRM Update on New Lead",
         projectId,
       })
       console.log("Flow created:", flow.id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       flow = client.workflows.create_flow(
           display_name="CRM Update on New Lead",
           project_id=project_id,
       )
       print("Flow created:", flow["id"])
       ```
     </TabItem>
   </Tabs>

3. **新增 Webhook 觸發器並發布流程**

   新建立的流程處於**草稿**狀態，沒有觸發器 — Webhook URL 還不存在，所以 `triggerFlow` 會回傳 404。先加入 Webhook 片段作為觸發器，然後發布：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 設定 Webhook 片段為流程的觸發器
       await client.workflows.applyFlowOperation(flow.id, {
         type: "UPDATE_TRIGGER",
         request: {
           name: "trigger",
           type: "PIECE_TRIGGER",
           valid: true,
           displayName: "Webhook",
           settings: {
             pieceName: "@activepieces/piece-webhook",
             pieceVersion: "0.1.24",
             triggerName: "catch_webhook",
             input: { authType: "none" },
             propertySettings: {},
           },
         },
       })

       // 發布 — 流程狀態從 DISABLED 變為 ENABLED，Webhook URL 生效
       await client.workflows.applyFlowOperation(flow.id, {
         type: "LOCK_AND_PUBLISH",
         request: {},
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 設定 Webhook 片段為流程的觸發器
       client.workflows.apply_flow_operation(flow["id"], {
           "type": "UPDATE_TRIGGER",
           "request": {
               "name": "trigger",
               "type": "PIECE_TRIGGER",
               "valid": True,
               "displayName": "Webhook",
               "settings": {
                   "pieceName": "@activepieces/piece-webhook",
                   "pieceVersion": "0.1.24",
                   "triggerName": "catch_webhook",
                   "input": {"authType": "none"},
                   "propertySettings": {},
               },
           },
       })

       # 發布 — DISABLED → ENABLED，Webhook URL 生效
       client.workflows.apply_flow_operation(flow["id"], {
           "type": "LOCK_AND_PUBLISH",
           "request": {},
       })
       ```
     </TabItem>
   </Tabs>

4. **使用負載手動觸發流程**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 發送後即忘（非同步）
       await client.workflows.triggerFlow(flow.id, {
         contact_name: "Jane Smith",
         email: "jane@example.com",
       })

       // 同步觸發 — 若要實際回傳資料而非超時，
       // 流程需要透過 applyFlowOperation ADD_ACTION 新增「Return Response」動作
       // （否則 Gateway 會等待 30 秒後放棄）。
       const result = await client.workflows.triggerFlowSync(flow.id, {
         contact_name: "Jane Smith",
         email: "jane@example.com",
       })
       console.log("Flow result:", result)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 發送後即忘（非同步）
       client.workflows.trigger_flow(flow["id"], {
           "contact_name": "Jane Smith",
           "email": "jane@example.com",
       })

       # 同步觸發 — 若要實際回傳資料而非超時，
       # 流程需要透過 apply_flow_operation ADD_ACTION 新增「Return Response」動作
       # （否則 Gateway 會等待 30 秒後放棄）。
       result = client.workflows.trigger_flow_sync(flow["id"], {
           "contact_name": "Jane Smith",
           "email": "jane@example.com",
       })
       print("Flow result:", result)
       ```
     </TabItem>
   </Tabs>

5. **將流程綁定至你的 AI Agent**

   在 Imbrace 儀表板中開啟你的 AI Agent，前往**工具 → 工作流程**，然後附加該流程。AI Agent 將能在對話中適時觸發它。

   或者，更新 AI Agent 使其透過名稱參考該工作流程：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       await client.chatAi.updateAiAgent(assistantId, {
         name: "Support Bot",
         workflow_name: "support_bot_v1",
         workflow_function_call: [{ flow_id: flow.id, description: "新潛在客戶時更新 CRM" }],
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       client.chat_ai.update_ai_agent(assistant_id, {
           "name": "Support Bot",
           "workflow_name": "support_bot_v1",
           "workflow_function_call": [
               {"flow_id": flow["id"], "description": "新潛在客戶時更新 CRM"}
           ],
       })
       ```
     </TabItem>
   </Tabs>

   <Tabs syncKey="lang">

## 3. 管理知識庫並附加至 AI Agent

知識庫檔案與資料夾位於 **data-board** 服務（`client.boards`）中。資料夾的 `_id` 即為你傳遞給 AI Agent 作為其知識來源的值。

<Steps>

1. **建立資料夾**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const folder = await client.boards.createFolder({
         name: "Product Documentation",
         organization_id: "org_your_org_id",
         parent_id: "root",
       })
       console.log("Folder ID:", folder._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       folder = client.boards.create_folder({
           "name": "Product Documentation",
           "organization_id": "org_your_org_id",
           "parent_id": "root",
       })
       print("Folder ID:", folder["_id"])
       ```
     </TabItem>
   </Tabs>

2. **上傳檔案至資料夾**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       import { readFileSync } from "fs"

       const fileBuffer = readFileSync("./docs/faq.pdf")
       const formData = new FormData()
       formData.append("file", new Blob([fileBuffer], { type: "application/pdf" }), "faq.pdf")
       formData.append("folder_id", folder._id)
       formData.append("organization_id", "org_your_org_id")

       const uploaded = await client.boards.uploadFile(formData)
       console.log("File uploaded:", uploaded.file_id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       from pathlib import Path

       path = Path("./docs/faq.pdf")
       files = {
           "file": (path.name, path.read_bytes(), "application/pdf"),
           "folder_id": (None, folder["_id"]),
           "organization_id": (None, "org_your_org_id"),
       }
       uploaded = client.boards.upload_file(files)
       print("File uploaded:", uploaded.get("file_id") or uploaded.get("_id"))
       ```
     </TabItem>
   </Tabs>

3. **將資料夾附加至 AI Agent**

   在 `folder_ids` 中傳入資料夾 `_id` — AI Agent 會從該資料夾中的所有檔案進行檢索。使用 `board_ids` 可附加 CRM 資料看板（其項目會成為知識來源 — 請參閱[資源 → 看板與項目](/sdk/resources/#boards--items--clientboards)）。舊版的 `knowledge_hubs` 欄位已棄用。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       await client.chatAi.updateAiAgent(assistantId, {
         name: "Support Bot",
         workflow_name: "support_bot_v1",
         folder_ids: [folder._id],
         // board_ids: [boardId],  // 可選：同時附加 CRM 資料看板
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       client.chat_ai.update_ai_agent(assistant_id, {
           "name": "Support Bot",
           "workflow_name": "support_bot_v1",
           "folder_ids": [folder["_id"]],
           # "board_ids": [board_id],  # 可選：同時附加 CRM 資料看板
       })
       ```
     </TabItem>
   </Tabs>

4. **檢查與管理資料夾與檔案**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 搜尋資料夾
       const folders = await client.boards.searchFolders({ q: "Product" })

       // 取得資料夾內容
       const contents = await client.boards.getFolderContents(folder._id)
       console.log("檔案數量：", contents.files?.length)

       // 重新命名資料夾
       await client.boards.updateFolder(folder._id, { name: "Product Docs v2" })

       // 搜尋資料夾中的檔案
       const files = await client.boards.searchFiles({ folderId: folder._id })

       // 刪除資料夾
       await client.boards.deleteFolders({ ids: [folder._id] })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 搜尋資料夾
       folders = client.boards.search_folders(q="Product")

       # 取得資料夾內容
       contents = client.boards.get_folder_contents(folder["_id"])
       print("檔案數量：", len(contents.get("files") or []))

       # 重新命名資料夾
       client.boards.update_folder(folder["_id"], {"name": "Product Docs v2"})

       # 搜尋資料夾中的檔案
       files = client.boards.search_files(folder_id=folder["_id"])

       # 刪除資料夾
       client.boards.delete_folders([folder["_id"]])
       ```
     </TabItem>
   </Tabs>

</Steps>

## 4. 管理資料看板與項目 (CRM 管道)

<Steps>

1. **建立看板**

   看板即 CRM 管道 — 潛在客戶、交易、任務或任何結構化資料。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const board = await client.boards.create({
         name: "Sales Pipeline",
         description: "追蹤所有進行中的交易",
       })
       console.log("Board ID:", board._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       board = client.boards.create(
           name="Sales Pipeline",
           description="追蹤所有進行中的交易",
       )
       print("Board ID:", board["_id"])
       ```
     </TabItem>
   </Tabs>

2. **新增自訂欄位**

   欄位類型包括 `ShortText`、`LongText`、`Number`、`Dropdown`、`Date`、`Checkbox` 等。`createField` 會回傳更新後的看板 — 在 `board.fields` 中找到你的新欄位。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const updated = await client.boards.createField(board._id, {
         name: "Company",
         type: "ShortText",
       })
       // 找出識別欄位（每個看板自動建立）
       const identifierField = updated.fields.find(f => f.is_identifier)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       updated = client.boards.create_field(board["_id"], {
           "name": "Company",
           "type": "ShortText",
       })
       identifier_field = next(f for f in updated["fields"] if f.get("is_identifier"))
       ```
     </TabItem>
   </Tabs>

3. **建立看板項目（記錄）**

   項目使用 `{ fields: [{ board_field_id, value }] }` 格式：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const item = await client.boards.createItem(board._id, {
         fields: [
           { board_field_id: identifierField._id, value: "Acme Corp" },
         ],
       })
       console.log("Item ID:", item._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       item = client.boards.create_item(board["_id"], {
           "fields": [
               {"board_field_id": identifier_field["_id"], "value": "Acme Corp"},
           ],
       })
       print("Item ID:", item["_id"])
       ```
     </TabItem>
   </Tabs>

4. **列出與搜尋項目**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 分頁瀏覽項目
       const { data: items } = await client.boards.listItems(board._id, { limit: 20, skip: 0 })

       // 全文搜尋
       const { data: results } = await client.boards.search(board._id, {
         q: "Acme",
         limit: 10,
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 分頁瀏覽項目
       items = client.boards.list_items(board["_id"], limit=20, skip=0)

       # 全文搜尋
       results = client.boards.search(board["_id"], q="Acme", limit=10)
       ```
     </TabItem>
   </Tabs>

   <Tabs syncKey="lang">

       await client.boards.updateItem(board._id, item._id, {
         fields: { name: "Acme Corp — Closed Won" },

     </TabItem>
     <TabItem label="Python">
       ```python

       client.boards.delete_item(board["_id"], item["_id"])
       ```
     </TabItem>
   </Tabs>

6. **匯出為 CSV**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const csv = await client.boards.exportCsv(board._id)
       // csv 為字串 — 可寫入檔案或做為下載傳送
       ```
     </TabItem>
     <TabItem label="Python">
