2. **创建 AI 代理**

   `workflow_name` 在你的组织中必须唯一。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const agent = await client.chatAi.createAiAgent({
         name: "Support Bot",
         workflow_name: "support_bot_v1",
         description: "处理一线客户支持查询",
         instructions: "你是一个乐于助人的支持代理。简洁且友好。",
         provider_id: "system",   // 使用组织的默认 LLM 提供商
         model_id: "gpt-4o",      // 系统提供商暴露的任何模型名称
       })

       const assistantId = agent.id  // UUID — 用于后续所有调用
       console.log("AI 代理已创建：", assistantId)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       agent = client.chat_ai.create_ai_agent({
           "name": "Support Bot",
           "workflow_name": "support_bot_v1",
           "description": "处理一线客户支持查询",
           "instructions": "你是一个乐于助人的支持代理。简洁且友好。",
           "provider_id": "system",   # 使用组织的默认 LLM 提供商
           "model_id": "gpt-4o",      # 系统提供商暴露的任何模型名称
       })

       assistant_id = agent["id"]
       print("AI 代理已创建：", assistant_id)
       ```
     </TabItem>
   </Tabs>

   `provider_id` 和 `model_id` 是必需的。传入 `provider_id: "system"` 委托给组织的默认 LLM 提供商，或传入自定义提供商的 UUID。使用 `provider_id: "system"` 时，`model_id` 接受类似 `"gpt-4o"` 的模型名称，或传入 `"Default"` 以回退到系统默认模型。

3. **使用 AI 代理流式接收聊天响应**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const response = await client.aiAgent.streamChat({
         assistant_id: assistantId,
         messages: [{ role: "user", content: "How do I reset my password?" }],
         // id 是会话 UUID — 重用它以保持对话历史
         // 如果省略，每次调用会自动生成一个新的 UUID
       })

       const reader = response.body!.getReader()
       const decoder = new TextDecoder()

       while (true) {
         const { done, value } = await reader.read()
         if (done) break
         const text = decoder.decode(value)
         for (const line of text.split("\n")) {
           if (line.startsWith("data: ")) {
             const data = line.slice(6).trim()
             if (data && data !== "[DONE]") {
               try {
                 const chunk = JSON.parse(data)
                 process.stdout.write(chunk.delta ?? chunk.content ?? "")
               } catch {}
             }
           }
         }
       }
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       response = client.ai_agent.stream_chat({
           "assistant_id": assistant_id,
           "messages": [{"role": "user", "content": "How do I reset my password?"}],
           # id 是会话 UUID — 重用它以保持对话历史。
           # 如果省略，每次调用会自动生成一个新的 UUID。
       })

       import json
       for line in response.iter_lines():
           if isinstance(line, bytes):
               line = line.decode()
           if not line.startswith("data: "):
               continue
           data = line[6:].strip()
           if data and data != "[DONE]":
               try:
                   chunk = json.loads(data)
                   print(chunk.get("delta") or chunk.get("content") or "", end="")
               except Exception:
                   pass
       ```
     </TabItem>
   </Tabs>

4. **维护对话历史（会话 ID）**

   在多次调用中传入相同的 `id`（必须是 UUID）以保持上下文：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       import { randomUUID } from "crypto"

       const sessionId = randomUUID()

       // 第一条消息
       await client.aiAgent.streamChat({
         assistant_id: assistantId,
         id: sessionId,
         messages: [{ role: "user", content: "What's your refund policy?" }],
       })

       // 跟进 — 同一会话，助手记住上下文
       await client.aiAgent.streamChat({
         assistant_id: assistantId,
         id: sessionId,
         messages: [{ role: "user", content: "How long does it take?" }],
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       import uuid

       session_id = str(uuid.uuid4())

       # 第一条消息
       client.ai_agent.stream_chat({
           "assistant_id": assistant_id,
           "id": session_id,
           "messages": [{"role": "user", "content": "What's your refund policy?"}],
       })

       # 跟进 — 同一会话，助手记住上下文
       client.ai_agent.stream_chat({
           "assistant_id": assistant_id,
           "id": session_id,
           "messages": [{"role": "user", "content": "How long does it take?"}],
       })
       ```
     </TabItem>
   </Tabs>

## 2. 创建工作流并将其绑定到助手

<Steps>

1. **列出已有流程以找到项目 ID**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const { data: flows } = await client.workflows.listFlows({ limit: 5 })
       const projectId = flows[0]?.projectId
       console.log("项目 ID：", projectId)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       res = client.workflows.list_flows(limit=5)
       flows = res.get("data", [])
       project_id = flows[0]["projectId"] if flows else None
       print("项目 ID：", project_id)
       ```
     </TabItem>
   </Tabs>

2. **创建新流程**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const flow = await client.workflows.createFlow({
         displayName: "新线索时更新 CRM",
         projectId,
       })
       console.log("流程已创建：", flow.id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       flow = client.workflows.create_flow(
           display_name="新线索时更新 CRM",
           project_id=project_id,
       )
       print("流程已创建：", flow["id"])
       ```
     </TabItem>
   </Tabs>

3. **添加 Webhook 触发器并发布流程**

   新创建的流程处于**草稿**状态，没有触发器——Webhook URL 尚不存在，因此 `triggerFlow` 会返回 404。将 Webhook 组件添加为触发器，然后发布：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 将 Webhook 组件设置为流程的触发器
       await client.workflows.applyFlowOperation(flow.id, {
         type: "UPDATE_TRIGGER",
         request: {
           name: "trigger",
           type: "PIECE_TRIGGER",
           valid: true,
           displayName: "Webhook",
           settings: {
             pieceName: "@activepieces/piece-webhook",
             pieceVersion: "0.1.24",
             triggerName: "catch_webhook",
             input: { authType: "none" },
             propertySettings: {},
           },
         },
       })

       // 发布 — 流程状态从 DISABLED 变为 ENABLED，Webhook URL 变为可用
       await client.workflows.applyFlowOperation(flow.id, {
         type: "LOCK_AND_PUBLISH",
         request: {},
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 将 Webhook 组件设置为流程的触发器
       client.workflows.apply_flow_operation(flow["id"], {
           "type": "UPDATE_TRIGGER",
           "request": {
               "name": "trigger",
               "type": "PIECE_TRIGGER",
               "valid": True,
               "displayName": "Webhook",
               "settings": {
                   "pieceName": "@activepieces/piece-webhook",
                   "pieceVersion": "0.1.24",
                   "triggerName": "catch_webhook",
                   "input": {"authType": "none"},
                   "propertySettings": {},
               },
           },
       })

       # 发布 — DISABLED → ENABLED，Webhook URL 变为可用
       client.workflows.apply_flow_operation(flow["id"], {
           "type": "LOCK_AND_PUBLISH",
           "request": {},
       })
       ```
     </TabItem>
   </Tabs>

4. **手动使用负载触发流程**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 触发并忘记（异步）
       await client.workflows.triggerFlow(flow.id, {
         contact_name: "Jane Smith",
         email: "jane@example.com",
       })

       // 同步触发 — 要让它实际返回数据而不超时，
       // 流程需要有一个通过 applyFlowOperation ADD_ACTION
       // 添加的 "Return Response" 动作（否则网关会等待 30 秒后放弃）。
       const result = await client.workflows.triggerFlowSync(flow.id, {
         contact_name: "Jane Smith",
         email: "jane@example.com",
       })
       console.log("流程结果：", result)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 触发并忘记（异步）
       client.workflows.trigger_flow(flow["id"], {
           "contact_name": "Jane Smith",
           "email": "jane@example.com",
       })

       # 同步触发 — 要让它实际返回数据而不超时，
       # 流程需要有一个通过 apply_flow_operation ADD_ACTION
       # 添加的 "Return Response" 动作（否则网关等待 30 秒后放弃）。
       result = client.workflows.trigger_flow_sync(flow["id"], {
           "contact_name": "Jane Smith",
           "email": "jane@example.com",
       })
       print("流程结果：", result)
       ```
     </TabItem>
   </Tabs>

5. **将流程绑定到 AI 代理**

   在 Imbrace 仪表盘中打开你的 AI 代理，进入**工具 → 工作流**，然后附加流程。AI 代理将能够在对话中适时触发它。

   或者，更新 AI 代理以通过名称引用工作流：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       await client.chatAi.updateAiAgent(assistantId, {
         name: "Support Bot",
         workflow_name: "support_bot_v1",
         workflow_function_call: [{ flow_id: flow.id, description: "新线索时更新 CRM" }],
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       client.chat_ai.update_ai_agent(assistant_id, {
           "name": "Support Bot",
           "workflow_name": "support_bot_v1",
           "workflow_function_call": [
               {"flow_id": flow["id"], "description": "新线索时更新 CRM"}
           ],
       })
       ```
     </TabItem>
   </Tabs>

## 3. 管理知识库并将其附加到 AI 代理

知识库文件和文件夹存在于 **data-board** 服务中（`client.boards`）。文件夹的 `_id` 就是你传递给 AI 代理作为其知识来源的内容。

<Steps>

1. **创建文件夹**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const folder = await client.boards.createFolder({
         name: "产品文档",
         organization_id: "org_your_org_id",
         parent_id: "root",
       })
       console.log("文件夹 ID：", folder._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       folder = client.boards.create_folder({
           "name": "产品文档",
           "organization_id": "org_your_org_id",
           "parent_id": "root",
       })
       print("文件夹 ID：", folder["_id"])
       ```
     </TabItem>
   </Tabs>

2. **上传文件到文件夹**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       import { readFileSync } from "fs"

       const fileBuffer = readFileSync("./docs/faq.pdf")
       const formData = new FormData()
       formData.append("file", new Blob([fileBuffer], { type: "application/pdf" }), "faq.pdf")
       formData.append("folder_id", folder._id)
       formData.append("organization_id", "org_your_org_id")

       const uploaded = await client.boards.uploadFile(formData)
       console.log("文件已上传：", uploaded.file_id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       from pathlib import Path

       path = Path("./docs/faq.pdf")
       files = {
           "file": (path.name, path.read_bytes(), "application/pdf"),
           "folder_id": (None, folder["_id"]),
           "organization_id": (None, "org_your_org_id"),
       }
       uploaded = client.boards.upload_file(files)
       print("文件已上传：", uploaded.get("file_id") or uploaded.get("_id"))
       ```
     </TabItem>
   </Tabs>

3. **将文件夹附加到 AI 代理**

   在 `folder_ids` 中传入文件夹 `_id`——AI 代理会检索该文件夹中的每个文件。使用 `board_ids` 附加 CRM 数据看板（其条目成为知识来源——参见[资源 → 看板与条目](/sdk/resources/#boards--items--clientboards)）。旧版 `knowledge_hubs` 字段已弃用。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       await client.chatAi.updateAiAgent(assistantId, {
         name: "Support Bot",
         workflow_name: "support_bot_v1",
         folder_ids: [folder._id],
         // board_ids: [boardId],  // 可选：同时附加 CRM 数据看板
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       client.chat_ai.update_ai_agent(assistant_id, {
           "name": "Support Bot",
           "workflow_name": "support_bot_v1",
           "folder_ids": [folder["_id"]],
           # "board_ids": [board_id],  # 可选：同时附加 CRM 数据看板
       })
       ```
     </TabItem>
   </Tabs>

4. **查看和管理文件夹与文件**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 搜索文件夹
       const folders = await client.boards.searchFolders({ q: "Product" })

       // 获取文件夹内容
       const contents = await client.boards.getFolderContents(folder._id)
       console.log("文件数量：", contents.files?.length)

       // 重命名文件夹
       await client.boards.updateFolder(folder._id, { name: "Product Docs v2" })

       // 搜索文件夹中的文件
       const files = await client.boards.searchFiles({ folderId: folder._id })

       // 删除文件夹
       await client.boards.deleteFolders({ ids: [folder._id] })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 搜索文件夹
       folders = client.boards.search_folders(q="Product")

       # 获取文件夹内容
       contents = client.boards.get_folder_contents(folder["_id"])
       print("文件数量：", len(contents.get("files") or []))

       # 重命名文件夹
       client.boards.update_folder(folder["_id"], {"name": "Product Docs v2"})

       # 搜索文件夹中的文件
       files = client.boards.search_files(folder_id=folder["_id"])

       # 删除文件夹
       client.boards.delete_folders([folder["_id"]])
       ```
     </TabItem>
   </Tabs>

</Steps>

## 4. 管理数据看板和条目（CRM 管道）

<Steps>

1. **创建看板**

   看板就是 CRM 管道——线索、交易、任务或任何结构化数据。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const board = await client.boards.create({
         name: "销售管道",
         description: "跟踪所有活跃交易",
       })
       console.log("看板 ID：", board._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       board = client.boards.create(
           name="销售管道",
           description="跟踪所有活跃交易",
       )
       print("看板 ID：", board["_id"])
       ```
     </TabItem>
   </Tabs>

2. **添加自定义字段**

   字段类型有 `ShortText`、`LongText`、`Number`、`Dropdown`、`Date`、`Checkbox` 等。`createField` 返回更新后的看板——在 `board.fields` 中找到你的新字段。

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const updated = await client.boards.createField(board._id, {
         name: "公司",
         type: "ShortText",
       })
       // 找到标识字段（每个看板自动创建）
       const identifierField = updated.fields.find(f => f.is_identifier)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       updated = client.boards.create_field(board["_id"], {
           "name": "公司",
           "type": "ShortText",
       })
       identifier_field = next(f for f in updated["fields"] if f.get("is_identifier"))
       ```
     </TabItem>
   </Tabs>

3. **创建看板条目（记录）**

   条目使用 `{ fields: [{ board_field_id, value }] }` 格式：

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       const item = await client.boards.createItem(board._id, {
         fields: [
           { board_field_id: identifierField._id, value: "Acme Corp" },
         ],
       })
       console.log("条目 ID：", item._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       item = client.boards.create_item(board["_id"], {
           "fields": [
               {"board_field_id": identifier_field["_id"], "value": "Acme Corp"},
           ],
       })
       print("条目 ID：", item["_id"])
       ```
     </TabItem>
   </Tabs>

4. **列出和搜索条目**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       // 分页获取条目
       const { data: items } = await client.boards.listItems(board._id, { limit: 20, skip: 0 })

       // 全文搜索
       const { data: results } = await client.boards.search(board._id, {
         q: "Acme",
         limit: 10,
       })
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       # 分页获取条目
       items = client.boards.list_items(board["_id"], limit=20, skip=0)

       # 全文搜索
       results = client.boards.search(board["_id"], q="Acme", limit=10)
       ```
     </TabItem>
   </Tabs>

5. **更新和删除条目**

   <Tabs syncKey="lang">
     <TabItem label="TypeScript">
       ```typescript
       await client.boards.updateItem(board._id, item._id, {
         fields: { name: "Acme Corp — 已赢单" },
       })

       await client.boards.deleteItem(board._id, item._id)
       ```
     </TabItem>
     <TabItem label="Python">
       ```python
       client.boards.update_item(board["_id"], item["_id"], {
           "fields": {"name": "Acme Corp — 已赢单"},
       })

       client.boards.delete_item(board["_id"], item["_id"])
       ```

     </TabItem>
   </Tabs>
